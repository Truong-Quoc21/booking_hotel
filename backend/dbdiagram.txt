Table users {
  id int [pk, increment]
  full_name varchar(100) [not null]
  email varchar(100) [not null, unique]
  password varchar(255) [not null]
  phone varchar(20)
  role varchar(20) [not null, note: 'admin | customer']
  status varchar(20) [not null, note: 'active | inactive | banned']
  created_at datetime [not null]
  updated_at datetime
}

Table hotels {
  id int [pk, increment]
  name varchar(255) [not null]
  description text
  address varchar(255) [not null]
  city varchar(100) [not null]
  country varchar(100)
  star_rating int
  phone varchar(20)
  email varchar(100)
  thumbnail varchar(255)
  check_in_time time
  check_out_time time
  created_at datetime [not null]
  updated_at datetime
  deleted_at datetime
}

Table hotel_images {
  id int [pk, increment]
  hotel_id int [not null]
  image_url varchar(255) [not null]
}

Table room_types {
  id int [pk, increment]
  type_name varchar(100) [not null]
  description text
}

Table rooms {
  id int [pk, increment]
  hotel_id int [not null]
  room_type_id int [not null]
  name varchar(100) [not null]
  price decimal(10,2) [not null]
  capacity int [not null]
  total_rooms int [not null]
  bed_count int
  area decimal(5,2)
  description text
  thumbnail varchar(255)
  status varchar(50) [not null, note: 'available | unavailable | maintenance']
  created_at datetime [not null]
  updated_at datetime
  deleted_at datetime
}

Table room_images {
  id int [pk, increment]
  room_id int [not null]
  image_url varchar(255) [not null]
}

Table amenities {
  id int [pk, increment]
  name varchar(100) [not null]
  icon varchar(255)
}

Table hotel_amenities {
  hotel_id int [not null]
  amenity_id int [not null]

  indexes {
    (hotel_id, amenity_id) [pk]
  }
}

Table bookings {
  id int [pk, increment]
  user_id int [not null]
  hotel_id int [not null]
  check_in_date date [not null]
  check_out_date date [not null]
  num_guests int [not null]
  total_amount decimal(10,2) [not null]
  booking_status varchar(50) [not null, note: 'pending | confirmed | cancelled | completed']
  payment_status varchar(50) [not null, note: 'unpaid | paid | refunded']
  special_request text
  created_at datetime [not null]
}

Table booking_details {
  id int [pk, increment]
  booking_id int [not null]
  room_id int [not null]
  quantity int [not null]
  price decimal(10,2) [not null]
}

Table payments {
  id int [pk, increment]
  booking_id int [not null]
  payment_method varchar(50) [not null, note: 'cash | bank_transfer | momo | vnpay | stripe']
  amount decimal(10,2) [not null]
  currency varchar(10) [not null, default: 'VND']
  transaction_code varchar(100)
  payment_status varchar(50) [not null, note: 'pending | success | failed | refunded']
  paid_at datetime
}

Table reviews {
  id int [pk, increment]
  user_id int [not null]
  hotel_id int [not null]
  booking_id int [not null]
  rating int [not null, note: '1 - 5']
  comment text
  created_at datetime [not null]

  indexes {
    (user_id, booking_id) [unique]
  }
}

Table wishlists {
  user_id int [not null]
  hotel_id int [not null]

  indexes {
    (user_id, hotel_id) [pk]
  }
}

Table vouchers {
  id int [pk, increment]
  code varchar(50) [not null, unique]
  discount_type varchar(20) [not null, note: 'percent | fixed']
  discount_value decimal(10,2) [not null]
  min_order_amount decimal(10,2)
  max_discount_amount decimal(10,2)
  start_date date [not null]
  end_date date [not null]
  quantity int [not null]
  used_count int [not null, default: 0]
  status varchar(50) [not null, note: 'active | inactive | expired']
}

Table booking_vouchers {
  booking_id int [not null]
  voucher_id int [not null]
  discount_amount decimal(10,2) [not null]

  indexes {
    (booking_id, voucher_id) [pk]
  }
}

Ref: reviews.hotel_id > hotels.id

Ref: wishlists.hotel_id > hotels.id
